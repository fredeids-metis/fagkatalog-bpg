Fagkatalog - Innbyggingskode

Kopier koden under og lim inn i en "Code Block" på Squarespace:

<!-- Fagkatalog Container -->
<div id="fagkatalog"></div>

<!-- Fagkatalog Loader -->
<script>
(function() {
    const v = Date.now();
    const base = 'https://fredeids-metis.github.io/fagkatalog-bpg';

    // Last CSS
    const link = document.createElement('link');
    link.rel = 'stylesheet';
    link.href = base + '/style.css?v=' + v;
    document.head.appendChild(link);

    // Bygg HTML-struktur
    document.getElementById('fagkatalog').innerHTML = `
        <main>
            <div class="filters">
                <div class="filters-row">
                    <input type="text" id="search" placeholder="Søk etter fag...">
                    <div class="filter-buttons">
                        <button class="filter-btn active" data-category="all">Alle</button>
                        <button class="filter-btn" data-category="realfag">Realfag</button>
                        <button class="filter-btn" data-category="spraak">Språk</button>
                        <button class="filter-btn" data-category="samfunnsfag">Samfunnsfag</button>
                    </div>
                </div>
            </div>
            <div id="fag-grid" class="fag-grid">
                <p class="loading">Laster fag...</p>
            </div>
        </main>
        <div id="modal" class="modal hidden">
            <div class="modal-content">
                <button class="close-btn" aria-label="Lukk">&times;</button>
                <div id="modal-body"></div>
            </div>
        </div>
    `;

    // Last app.js
    const script = document.createElement('script');
    script.src = base + '/app.js?v=' + v;
    document.body.appendChild(script);
})();
</script>
